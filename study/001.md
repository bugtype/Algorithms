```py
answer = False

def solution():
  global answer

  teacherCount = 0

  def pre_processing():
    for (x, y) in ts:
      if find_student(x, y) == True:
        return False
    return True

  def find_student(x, y):
    moveX = [0, 0, 1, -1]
    moveY = [1, -1, 0, 0]
    for i in range(4):
      xx = x+moveX[i]
      yy = y+moveY[i]
      while 0 <= xx and xx < n and 0 <= yy and yy < n and arr[xx][yy] != 'O':
        if arr[xx][yy] == 'S':
          return True
        else:
          xx += moveX[i]
          yy += moveY[i]
    return False

  def install_obstacle(x, y, count):
    global answer
    if count == 4:
      return
    if count < 3:
      for i in range(x, n):
        for j in range(n):
          if arr[i][j] == 'X':
            arr[i][j] = 'O'
            count += 1
            install_obstacle(i, j, count)
            count -= 1
            arr[i][j] = 'X'
    # if 3
    cnt = 0
    for i in range(n):
      for j in range(n):
        if arr[i][j] == 'T':
          if find_student(i, j) == False:
            cnt += 1
    if cnt == teacherCount:
      answer = True
    return

  n = int(input())
  arr = []
  for _ in range(n):
    data = list(input().strip().split(' '))
    arr.append(data)

  # 어허 설치안해도 피할수 있다!
  ts = [(i, j) for i in range(len(arr))
        for j in range(len(arr[i])) if arr[i][j] == 'T']
  teacherCount = len(ts)
  if pre_processing():
    print("YES")
    return

  install_obstacle(0, 0, 0)

  if answer:
    print('YES')
  else:
    print('NO')


solution()
```
